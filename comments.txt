def analyze_type(character):
    if character in digit:
        print("type: digit")
        return "digit"
    elif character in letter:
        print("type: letter")
        return "letter"
    elif character in math:
        print("type: math")
        return "math"
    elif character in relational:
        print("type: relational")
        return "relational"
    elif character == space:
        print("type: space")
        return "space"
    else:   
        print("type: unrecognized")
        return "unrecognized"
